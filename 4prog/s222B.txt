program s222B;
var
  s: string;
  c: char;
  i, j, n, plen: integer;
  haveprint: boolean;
begin
	for i:= 1 to ParamCount do
	begin
		haveprint := true;
		s := ParamStr(i);
		plen := Length(s);
		for j := 1 to plen do
		begin
			c := s[i];
			for n := i + 1 to plen do
			begin
				if c = s[n] then
					haveprint := false
			end;
		end;
		if haveprint then
			writeln(ParamStr(i));
	end
end.
